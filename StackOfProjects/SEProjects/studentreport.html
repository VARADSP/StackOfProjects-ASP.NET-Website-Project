		void student::calculate()
		{
			per=(p_marks+c_marks+m_marks+e_marks+cs_marks)/5.0;
			if(per>=60)
				grade='A';
			else if(per>=50)
				grade='B';
			else if(per>=33)
				grade='C';
			else
				grade='F';
		}
